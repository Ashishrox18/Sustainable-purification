let simInterval = null;
const phE2 = document.getElementById('ph');
const tdsEl = document.getElementById('tds');
const tempE2 = document.getElementById('temp');
const moistEl = document.getElementById('moist');
const recText = document.getElementById('recText');

function randomBetween(min, max, fixed=1) {
  return (Math.random() * (max - min) + min).toFixed(fixed);
}

function simulateStep(){
  // Simulate sensor readings
  const ph_value = randomBetween(8.5, 10, 2);
  const tds = Math.round(randomBetween(150, 850, 0)); // ppm
  const temp = randomBetween(20, 34, 1);
  const moist = Math.round(randomBetween(18, 86, 0)); // %

  phE2.textContent = ph;
  tdsEl.textContent = tds;
  tempEl.textContent = temp;
  moistEl.textContent = moist;


  let msgs = [];
  if (ph < 6.5) msgs.push('Increase alkalinity: consider liming before irrigation.');
  if (ph > 8.2) msgs.push('High pH: check source water mixing or buffer before use.');
  if (tds > 1000) msgs.push('High salinity: dilute treated water or use for tolerant crops only.');
  if (moist < 30) msgs.push('Soil dry: schedule irrigation (200â€“300 L/acre now).');
  if (moist > 75) msgs.push('Soil saturated: delay irrigation to avoid waterlogging.');

  if (msgs.length === 0) {
    recText.textContent = 'Water and soil parameters look stable. Maintain current schedule and monitor daily.';
  } else {
    recText.textContent = msgs.join(' ');
  }
}

document.getElementById('startSim').addEventListener('click', () => {
  if (simInterval) return;
  simulateStep();
  simInterval = setInterval(simulateStep, 2500);
});

document.getElementById('stopSim').addEventListener('click', () => {
  if (!simInterval) return;
  clearInterval(simInterval);
  simInterval = null;
});

document.getElementById('resetSim').addEventListener('click', () => {
